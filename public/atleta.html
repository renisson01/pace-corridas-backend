async function carregarTreinoHoje() { try { const res = await fetch('/coach/daily', { headers: { Authorization: `Bearer ${localStorage.getItem('token')}` } }); const data = await res.json(); if (data.treino) { renderTreino(data.treino); } else { document.getElementById('treino-hoje').innerHTML = 'Nenhum treino hoje. Faça recuperação leve.'; } } catch (e) { document.getElementById('treino-hoje').innerHTML = 'Erro ao carregar treino.'; } }
